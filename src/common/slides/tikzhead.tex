% lattice that can be dropped into any path.
% it would be sensible to set a default argument here, but
% commands with optional args don't parse correctly inside a
% tikz path
\newcommand*{\latt}[1]{%
  \foreach \a in {-4, ..., 4} {
    \foreach \b in {-4, ..., 4} {
      (\a,\b) circle (#1)
    }
  }
}
